JS code sample

```
function solution(matrix) {
    const rowsNumber = matrix.length; 
    const colsNumber = matrix[0].length;
    const res = [];
    
    for (let i = 0; i < colsNumber; i++) {
        for (let j = 0; j < rowsNumber; j++) {
            if (matrix[j][i] === 0) {
                break;
            } else {
                res.push(matrix[j][i]);
            }
        }
    }
    
    return res.reduce((total, item) => total += item, 0);
}
```
